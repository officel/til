    > sudo docker build -t gatling .

    sudo docker ps -a
    sudo docker cp 2182d097856b:/gatling/results/ /tmp/
    sudo docker run --rm -d -p 8080:80 -v `pwd`:/usr/share/nginx/html nginx:alpine
    sudo docker ps
    sudo docker stop 4d831eed4683
